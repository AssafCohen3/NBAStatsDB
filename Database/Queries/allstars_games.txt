select SEASON, GAME_ID, GAME_DATE, MATCHUP, count(distinct A.PLAYER_ID) as ALLSTARS_COUNT, group_concat(A.PLAYER_NAME) as ALLSTARS from (
    select SEASON, GAME_ID, GAME_DATE, MATCHUP, AC.PLAYER_ID, AC.PLAYER_NAME from BoxScoreP inner join AllstarCareer AC on BoxScoreP.PLAYER_ID = AC.PLAYER_ID
    where SEASON_TYPE != 3
    ) A
group by GAME_ID
order by ALLSTARS_COUNT desc
